CREATE TABLE Items(
  ID serial,
  Name varchar(256) NOT NULL,
  TemplateID bigint,
  ParentID bigint,
  Created datetime NOT NULL,
  Updated datetime NOT NULL
);


CREATE TABLE Fields(
  Id SERIAL,
  ItemId bigint unique,
  Language varchar(50),
  Version int,
  FieldId bigint unique,
  Value varchar(58),
  Created datetime,
  Updated datetime
);
